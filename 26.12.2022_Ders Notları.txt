create database HASTANE

CREATE TABLE Hasta(
HastaId int not null Primary Key,
Soyad nvarchar(20) not null,
Ad nvarchar(20) not null,
Cinsiyet nchar(1) not null,
Boy float not null,
Kilo float not null
)
